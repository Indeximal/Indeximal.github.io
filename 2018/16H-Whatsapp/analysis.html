import re
import datetime

class Message:
    def __init__(self, date, time, sender, msg):
        self.date = date
        self.time = time
        self.sender = sender
        self.msg = msg
    def getDatetime(self):
        return datetime.datetime(2000+ int(self.date[6:]), int(self.date[3:5]), int(self.date[:2]), int(self.time[:2]), int(self.time[3:5]), int(self.time[6:]))
    def getType(self):
        if "PHOTO" in self.msg or "Bild weggelassen" in self.msg:
            return "image"
        if "VIDEO" in self.msg or "Video weggelassen" in self.msg:
            return "video"
        if "GIF" in self.msg or "GIF weggelassen" in self.msg:
            return "gif"
        if "AUDIO" in self.msg or "Audio weggelassen" in self.msg:
            return "audio"
        if "<angehngt>" in self.msg or "<Anhang" in self.msg or "Dokument weggelassen" in self.msg:
            return "document"
        if "Sticker weggelassen" in self.msg:
            return "sticker"
        return "text"

%matplotlib inline
import matplotlib.pyplot as plt
import numpy as np

spectralColorMap = plt.get_cmap("Spectral")

def getColors(n):
    return spectralColorMap(np.linspace(0, 1, n))

def getValueColors(arr):
    na = np.array(arr)
    return spectralColorMap(1-(na - na.min())/(na.max() - na.min()))

def getColor():
    return spectralColorMap(0)

def plotBarChart(values, labels, title=None):
    xcoords = np.arange(len(values))
    plt.figure(figsize=(16, 4))
    plt.grid(True, axis="y")
    plt.bar(xcoords, values, color=getValueColors(values))
    plt.xticks(xcoords, labels, rotation=90)
    plt.title(title)
    
def plotPieChart(values, labels, title=None, autopct="%1.1f%%"):
    plt.figure(figsize=(10, 7))
    plt.pie(values, labels=labels, startangle=90, counterclock=False, pctdistance=0.75, 
            autopct=autopct, colors=getColors(len(values)), wedgeprops=dict(edgecolor='w'))
    plt.title(title)
